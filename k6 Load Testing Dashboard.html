// This is a sample k6 script that you can use for testing import http from 'k6/http'; import { sleep, check } from 'k6'; export const options = { vus: 5, duration: '30s', }; const API_BASE_URL = 'https://reqres.in/api'; const USERNAME = 'eve.holt@reqres.in'; const PASSWORD = 'cityslicka'; export default function () { // Login request const loginRes = http.post(`${API_BASE_URL}/login`, { email: USERNAME, password: PASSWORD, }); check(loginRes, { 'login status is 200': (r) => r.status === 200, }); // Get user details const userRes = http.get(`${API_BASE_URL}/users/2`); check(userRes, { 'user status is 200': (r) => r.status === 200, }); sleep(1); }
